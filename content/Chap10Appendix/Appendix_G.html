The Fourier Transform#

Definitions#

(508)#\[\begin{align*} {\cal F}(h)(\xi,\eta) &= \int\!\!\!\int e^{-2\pi i(x \xi + y \eta)} h(x,y) dx\, dy. \end{align*}\]
(509)#\[\begin{split}\begin{align*} \\ {\cal F}^{-1}(H)(x,y) &= \int\!\!\!\int e^{2\pi i(x \xi + y \eta)} H(\xi,\eta) d \xi \, df_y.\end{align*}\end{split}\]

General Equations#

(510)#\[\begin{align*} {\cal F}^{-1}{\cal F}(h)(x,y) &= h(x,y), \end{align*}\]
(511)#\[\begin{split}\begin{align*} \\ {\cal F}(h)(\xi,\eta)^* &= {\cal F}(h^*)(-\xi,-\eta),\end{align*}\end{split}\]

(\(z^*\) is the complex conjugate of \(z\)).

(512)#\[\begin{align*} \int\!\!\!\int | h(x,y)|^2 dx\, dy &= \int\!\!\!\int |{\cal F}(h)(\xi,\eta)|^2 d \xi\, df_y, \text{ Parseval's formula}), \end{align*}\]
(513)#\[\begin{split}\begin{align*} \\ {\cal F}(g*h) &= {\cal F}(g) {\cal F}(h), \end{align*}\end{split}\]
(514)#\[\begin{split}\begin{align*} \\ {\cal F}(gh) &= {\cal F}(g) * {\cal F}(h),\end{align*}\end{split}\]

where

(515)#\[\begin{align*} (g*h)(x,y) = \int\!\!\!\int g(x-x', y-y') h(x',y') dx' \, dy'. \end{align*}\]

If \(h(x)\) is a \(p\)-periodical function then

(516)#\[\begin{align*} {\cal F}(h)(\xi) = \sum_{n=-\infty}^{+\infty} \hat{h}(n)\, \delta\left( \xi - \frac{n}{p}\right), \end{align*}\]

where

(517)#\[\begin{align*} \hat{h}(n) = \frac{1}{p} \int_0^p h(x) e^{-2\pi n x} \, dx. \end{align*}\]

Some Fourier transforms#

(518)#\[\boxed{\begin{align*} {\cal F}\left[ 1_{[-a,a]}(x) 1_{[-b,b]} \right](\xi, \eta) = 4 a b \, \text{sinc}(2 a \xi) \text{sinc}(2bf_y), \end{align*}}\]

where

(519)#\[\begin{align*} \text{sinc} (x) = \frac{\sin (\pi x)}{\pi x}. \end{align*}\]
(520)#\[\boxed{\begin{align*} {\cal F}\left[ \delta(x/a) \delta(y/b)\right] = a b. \end{align*}}\]
(521)#\[\boxed{\begin{align*} {\cal F}\left[1\right](\xi,\eta) = \delta(\xi)\delta(\eta). \end{align*}}\]
(522)#\[\boxed{\begin{align*} {\cal F}\left[ e^{-\pi(a^2 x^2 + b^2 y^2})\right] (\xi,\eta) = \frac{1}{|ab|} e^{-\pi\left(\xi^2/a^2 + \eta^2/b^2 \right) }. \end{align*}}\]

Let

(523)#\[\begin{split}\begin{align*} 1_{\bigcirc_a}(x,y) = \left\{ \begin{array}{l}1, \;\; \text{ als } \;\\\sqrt{x^2 + y^2} \leq a, \\0, \;\; \text{ als } \; \sqrt{x^2+y^2} > a. \end{array} \right. \end{align*}\end{split}\]

Then

(524)#\[\boxed{\begin{align*} {\cal F}(1_{\bigcirc_a})(\xi,\eta) =a \frac{J_1 \left( 2\pi a \sqrt{\xi^2+\eta^2}\right)} { \sqrt{\xi^2 + \eta^2}}. \end{align*}}\]
(525)#\[\boxed{\begin{align*} {\cal F}\left[ e^{i \pi\left( a ^2 x^2 + b^2 y^2 \right)}\right](\xi,\eta) = \frac{i}{|ab|} e^{-i\pi\left( \xi^2/a^2 + \eta^2/b^2\right)} \end{align*}}\]
